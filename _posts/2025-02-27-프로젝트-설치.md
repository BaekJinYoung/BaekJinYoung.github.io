
다음 내용을 install.sh 파일에 붙여넣기하세요.
작성 후에는 Ctrl + x 키를 누른 후 y를 입력해 저장합니다.

```shell
#!/bin/bash

set -e

sudo apt-get clean --assume-yes
sudo apt-get update --assume-yes
sudo apt-get upgrade --assume-yes

# Composer 설치 함수
install_composer() {
    curl -sS https://getcomposer.org/installer | php
    sudo mv composer.phar /usr/local/bin/composer
    sudo chmod +x /usr/local/bin/composer
}

# Laravel 설치 함수
install_laravel() {
    sudo composer global require laravel/installer
    sudo ln -s ~/.composer/vendor/bin/laravel /usr/local/bin/laravel
}

# 필수 패키지 설치 함수
install_packages() {
    sudo apt install software-properties-common
    sudo add-apt-repository ppa:ondrej/php
    sudo apt update

    sudo apt-get install -y \
        curl \
        git \
        unzip \
        zip \
        nginx \
        mysql-server \
        mysql-client \
        php8.4-cli php8.4-fpm php8.4-mysql php8.4-xml php8.4-mbstring php8.4-curl php8.4-zip php8.4-gd php8.4-bcmath php8.4-sqlite3 \
        phpmyadmin \
        vsftpd \
        certbot \
        python3-certbot-nginx
}

# Configure firewall
configure_firewall() {
    sudo ufw allow OpenSSH
    sudo ufw allow mysql
    sudo ufw allow 80/tcp
    sudo ufw allow 443/tcp
    sudo ufw allow 2121/tcp
    sudo ufw allow 20:21/tcp
    sudo ufw allow 10090:10100/tcp
    sudo ufw allow 30000:31000/tcp
    sudo ufw --force enable
}

# Main function to execute all configuration steps
main() {
    install_packages
    install_composer
    install_laravel
    configure_firewall
}

main
```

해당 파일을 실행시킵니다.

```shell
sudo -u username bash /home/username/install.sh
```

Nginx 설정
-

```shell
sudo nano /etc/nginx/sites-available/default
```

아래와 같은 내용으로 수정합니다.
주석을 참고해주세요.
작성 후에는 Ctrl + x 키를 누른 후 y를 입력해 저장합니다.

```shell
server {

        root /home/username/laravel/public;

        index index.php;

        # 언더바(_)를 지운 뒤 도메인 주소를 입력합니다. 
        # 예시: server_name www.naver.com;
        server_name _; 

        location / {
          try_files $uri $uri/ /index.php?$query_string;
        }

        client_max_body_size 500M;

        # 설치한 php-fpm 버전으로 설정해야 합니다.
        # 보내드린 스크립트와 같은 버전이니 수정하지 않으셔도 됩니다.
        location ~ \.php$ {
          include snippets/fastcgi-php.conf;

          fastcgi_pass unix:/run/php/php8.4-fpm.sock;
        }

        location ~ /\.(?!well-known).* {
          deny all;
        }
}

# 이하는 건들지 않으셔도 됩니다.
server {
       listen 80;
       listen [::]:80;

       server_name example.com;

       root /var/www/example.com;
       index index.html;

       location / {
          try_files $uri $uri/ =404;
       }
}
```

```shell
sudo systemctl restart nginx
```
